手写apply，bind，call函数

手写call

Function.prototype.newCall=function(ctx,...args){
        // 将方法挂载到我们传入的ctx
        ctx.fn=this;
        // 将挂载 以后的方法调用
        ctx.fn(...args)
        // 将我们添加的这个属性删除
        delete  ctx.fn
    }
    function show(){
        console.log(this.name);
        console.log(...arguments)
    }
    show.newCall({name:'1111'},'12313','1654654654')

手写bind

Function.prototype.newBind=function(ctx,...args1){

        return (...args2)=>{
            ctx.fn =this;
            ctx.fn(...args1.concat(args2))
            delete ctx.fn
        }
    }
    function show(){
        console.log(...arguments)
        console.log(this.name);
    }
    let bind =show.newBind({name:'1'},'1111','22222')
    bind('sjds')

手写apply

Function.prototype.newApply=function(ctx,args=[]){
        // 判断是否为数组
        if(args && !(args instanceof Array)){
            throw ('只接受数组')
        }
        // 将fn方法挂载到传入的对象中
        ctx.fn =this
        // 调用fn 方法
        ctx.fn(...args)
        delete ctx.fn
    }
    function show(){
        console.log(this.name);
        console.log(...arguments);
    }
    show.newApply({name:'1111'},'kdja')
